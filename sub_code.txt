def _json_safe(v):
    """Convert values to JSON-safe types (replace NaN/Inf with None)."""
    if v is None:
        return None
    if hasattr(v, "item") and callable(getattr(v, "item")):
        try:
            v = v.item()
        except Exception:
            pass
    if isinstance(v, (float, np.floating)):
        if math.isnan(float(v)) or math.isinf(float(v)):
            return None
        return float(v)
    return v

# Keep POIs in EPSG:4326 for payloads
_poi_geoms = list(pois.geometry)
_poi_tree = STRtree(_poi_geoms)

# Also build a meter-based index (EPSG:3857) for accurate distance filtering (e.g., 20m rule)
_pois_3857 = pois.to_crs("EPSG:3857")
_poi_geoms_3857 = list(_pois_3857.geometry)
_poi_tree_3857 = STRtree(_poi_geoms_3857)
# Map geometry object id -> row index for fast lookup when STRtree returns geometries
_poi_geom_id_to_idx_3857 = {id(g): i for i, g in enumerate(_poi_geoms_3857)}


def _nearest_poi_index(lat: float, lon: float) -> int:
    q = Point(lon, lat)
    nearest = _poi_tree.nearest(q)
    if nearest is None:
        raise HTTPException(404, "No POI found.")
    if isinstance(nearest, (int, np.integer)):
        return int(nearest)
    return _poi_geoms.index(nearest)


def _hits_to_indices(hits, id_to_idx):
    """Convert STRtree query results to integer indices (supports Shapely 1.x geometries or Shapely 2.x indices)."""
    out = []
    for h in hits:
        if isinstance(h, (int, np.integer)):
            out.append(int(h))
        else:
            idx = id_to_idx.get(id(h))
            if idx is not None:
                out.append(int(idx))
    return out


def _poi_candidates_within_20m(lat: float, lon: float, limit: int = 10):
    """Return POI row indices within 20m of (lat, lon), sorted by distance ascending (meters)."""
    limit = max(1, min(50, int(limit)))

    q_ll = Point(lon, lat)
    q_3857 = gpd.GeoSeries([q_ll], crs="EPSG:4326").to_crs("EPSG:3857").iloc[0]

    # Query POIs within a 20m buffer
    buf = q_3857.buffer(20.0)
    hits = _poi_tree_3857.query(buf)
    idxs = _hits_to_indices(hits, _poi_geom_id_to_idx_3857)
    if not idxs:
        return []

    # Compute distances in meters and sort
    dists = []
    for idx in set(idxs):
        g = _poi_geoms_3857[idx]
        dists.append((float(q_3857.distance(g)), int(idx)))

    dists.sort(key=lambda x: x[0])
    return [idx for _, idx in dists[:limit]]


def _poi_payload(idx: int):
    row = pois.iloc[idx]
    props = {}
    for k, v in row.drop(labels=["geometry"]).to_dict().items():
        props[k] = _json_safe(v)
    props["poi_latitude"] = _json_safe(float(row.geometry.y))
    props["poi_longitude"] = _json_safe(float(row.geometry.x))
    return props


def _distance_m_approx(lat1: float, lon1: float, lat2: float, lon2: float) -> float:
    g = gpd.GeoSeries([Point(lon1, lat1), Point(lon2, lat2)], crs="EPSG:4326").to_crs("EPSG:3857")
    return float(g.iloc[0].distance(g.iloc[1]))


class RouteReq(BaseModel):
    start_lat: float
    start_lon: float

    fallback_to_raw: bool = True  # if POI-snapped routing fails, retry using raw coordinates
    poi_candidates: int = 3        # max POI candidates (within 20m) to try for start/end

@app.get("/graph_stats")
def graph_stats():
    """Basic stats for the routing graph to diagnose disconnected networks."""
    G = getattr(engine, "G", None)
    if G is None:
        return {"ok": False, "error": "RouteEngine has no attribute 'G'"}

    # connected components (undirected view)
    Gu = G.to_undirected(as_view=True)
    comps = list(nx.connected_components(Gu))
    comp_sizes = sorted([len(c) for c in comps], reverse=True)

    return {
        "ok": True,
        "nodes": int(G.number_of_nodes()),
        "edges": int(G.number_of_edges()),
        "components": int(len(comps)),
        "largest_component_nodes": int(comp_sizes[0]) if comp_sizes else 0,
        "top5_component_sizes": comp_sizes[:5],
    }


@app.get("/roads_bounds")
def roads_bounds():
    minx, miny, maxx, maxy = engine.roads.total_bounds
    pts = gpd.GeoSeries([Point(minx, miny), Point(maxx, maxy)], crs="EPSG:3857").to_crs("EPSG:4326")
    pmin, pmax = pts.iloc[0], pts.iloc[1]
    return {
        "crs": "EPSG:4326",
        "min_lon": float(pmin.x),
        "min_lat": float(pmin.y),
        "max_lon": float(pmax.x),
        "max_lat": float(pmax.y),
    }


@app.get("/suggest_points")
def suggest_points(n: int = 5):
    n = max(1, min(50, n))
    sampled = []
    roads_ll = engine.roads.to_crs("EPSG:4326")
    step = max(1, len(roads_ll) // n)
    for i in range(0, len(roads_ll), step):
        if len(sampled) >= n:
            break
        geom = roads_ll.geometry.iloc[i]
        if geom is None or geom.is_empty:
            continue
        try:
            p = geom.interpolate(0.5, normalized=True)
        except Exception:
            continue
        sampled.append({"lon": float(p.x), "lat": float(p.y)})
    return {"count": len(sampled), "points": sampled}


@app.get("/suggest_pair")
def suggest_pair():
    """Return a start/end pair sampled from the SAME road geometry (EPSG:4326) for reliable routing tests."""
    roads_ll = engine.roads.to_crs("EPSG:4326")
    for i in range(len(roads_ll)):
        geom = roads_ll.geometry.iloc[i]
        if geom is None or geom.is_empty:
            continue
        # handle MultiLineString by taking the longest part
        try:
            if geom.geom_type == "MultiLineString":
                geom = max(list(geom.geoms), key=lambda g: g.length)
        except Exception:
            pass

        try:
            p1 = geom.interpolate(0.25, normalized=True)
            p2 = geom.interpolate(0.75, normalized=True)
        except Exception:
            continue

        start = {"lon": float(p1.x), "lat": float(p1.y)}
        end = {"lon": float(p2.x), "lat": float(p2.y)}

        # Also show nearest POIs to these points (useful for POI-snap debugging)
        try:
            s_idx = _nearest_poi_index(start["lat"], start["lon"])
            e_idx = _nearest_poi_index(end["lat"], end["lon"])
            start_poi = _poi_payload(s_idx)
            end_poi = _poi_payload(e_idx)
        except Exception:
            start_poi = None
            end_poi = None

        return {
            "start": start,
            "end": end,
            "nearest_poi": {"start": start_poi, "end": end_poi},
        }

    raise HTTPException(404, "No suitable road geometry found to suggest a pair.")

@app.post("/route_top3")
def route_top3(req: RouteReq):
    mode = req.mode.strip().lower()
    if mode not in ("car", "motorcycle"):
        raise HTTPException(400, "mode must be 'car' or 'motorcycle'")

    print(
        f"[route_tool] /route_top3 start | mode={mode} k={req.k} snap_to_poi={req.snap_to_poi} "
        f"fallback_to_raw={req.fallback_to_raw} poi_candidates={req.poi_candidates}"
    )

    try:
        def _try_route(sl: float, so: float, el: float, eo: float):
            print(f"[route_tool] full route try | start=({sl:.6f},{so:.6f}) end=({el:.6f},{eo:.6f}) k={req.k}")
            return engine.route_top3(
                so,
                sl,
                eo,
                el,
                mode=mode,
                k=req.k,
            )

        def _try_route_fast(sl: float, so: float, el: float, eo: float):
            # Lightweight route existence check while evaluating POI candidate pairs
            print(f"[route_tool] fast route try | start=({sl:.6f},{so:.6f}) end=({el:.6f},{eo:.6f})")
            return engine.route_top3(
                so,
                sl,
                eo,
                el,
                mode=mode,
                k=1,
            )

        used = "raw"
        snap_info = None

        # Pass 1: POI-snapped routing (strict 20m rule + try alternates within 20m)
        if req.snap_to_poi:
            cand_limit = int(req.poi_candidates)
            s_cands = _poi_candidates_within_20m(req.start_lat, req.start_lon, limit=cand_limit)
            e_cands = _poi_candidates_within_20m(req.end_lat, req.end_lon, limit=cand_limit)

            # If either side has no POI within 20m, skip POI snapping entirely
            if not s_cands or not e_cands:
                start_lat, start_lon = float(req.start_lat), float(req.start_lon)
                end_lat, end_lon = float(req.end_lat), float(req.end_lon)
                print("[route_tool] fallback to raw | reason=no_poi_within_20m")
                routes = _try_route(start_lat, start_lon, end_lat, end_lon)
                used = "raw" if routes else used
                snap_info = {
                    "rule": "20m",
                    "status": "skipped",
                    "reason": "no_poi_within_20m",
                    "start_has_poi": bool(s_cands),
                    "end_has_poi": bool(e_cands),
                }
            else:
                # Build sorted candidate POIs with distances for transparency
                def _cand_info(lat, lon, idxs):
                    out = []
                    for idx in idxs:
                        p = _poi_payload(idx)
                        d = _distance_m_approx(lat, lon, float(p["poi_latitude"]), float(p["poi_longitude"]))
                        out.append({"idx": int(idx), "dist_m": float(d), "poi": p})
                    out.sort(key=lambda x: x["dist_m"])
                    return out

                s_infos = _cand_info(req.start_lat, req.start_lon, s_cands)
                e_infos = _cand_info(req.end_lat, req.end_lon, e_cands)

                # Preselect the nearest candidate on each side for a fast first attempt
                nearest_start = s_infos[0] if s_infos else None
                nearest_end = e_infos[0] if e_infos else None

                snap_info = {
                    "rule": "20m",
                    "status": "attempted",
                    "start": {"input": {"lat": float(req.start_lat), "lon": float(req.start_lon)}, "candidates": s_infos},
                    "end": {"input": {"lat": float(req.end_lat), "lon": float(req.end_lon)}, "candidates": e_infos},
                }

                routes = []
                chosen = None

                # Fast first attempt: nearest start POI + nearest end POI
                if nearest_start is not None and nearest_end is not None:
                    sl = float(nearest_start["poi"]["poi_latitude"])
                    so = float(nearest_start["poi"]["poi_longitude"])
                    el = float(nearest_end["poi"]["poi_latitude"])
                    eo = float(nearest_end["poi"]["poi_longitude"])
                    routes = _try_route_fast(sl, so, el, eo)
                    if routes:
                        chosen = {"start": nearest_start, "end": nearest_end, "pairs_tried": 1}
                        start_lat, start_lon = sl, so
                        end_lat, end_lon = el, eo
                        used = "poi"

                # Try combinations, closest-first (small bounded search)
                max_pairs = 4
                pairs_tried = 0 if not chosen else 1
                for si in s_infos:
                    if chosen:
                        break
                    for ei in e_infos:
                        pairs_tried += 1
                        if pairs_tried > max_pairs:
                            break
                        sl = float(si["poi"]["poi_latitude"])
                        so = float(si["poi"]["poi_longitude"])
                        el = float(ei["poi"]["poi_latitude"])
                        eo = float(ei["poi"]["poi_longitude"])
                        routes = _try_route_fast(sl, so, el, eo)
                        if routes:
                            chosen = {"start": si, "end": ei, "pairs_tried": pairs_tried}
                            start_lat, start_lon = sl, so
                            end_lat, end_lon = el, eo
                            used = "poi"
                            break
                    if routes or pairs_tried > max_pairs:
                        break

                if chosen:
                    snap_info["status"] = "used_poi"
                    snap_info["chosen"] = chosen
                    # Rerun full routing for the chosen POI pair using the requested k
                    routes = _try_route(start_lat, start_lon, end_lat, end_lon)
                else:
                    snap_info["status"] = "failed_poi"
                    snap_info["pairs_tried"] = pairs_tried
                    # Pass 2: fallback to raw coordinates if enabled
                    if req.fallback_to_raw:
                        start_lat, start_lon = float(req.start_lat), float(req.start_lon)
                        end_lat, end_lon = float(req.end_lat), float(req.end_lon)
                        print("[route_tool] fallback to raw | reason=failed_poi_candidates")
                        routes = _try_route(start_lat, start_lon, end_lat, end_lon)
                        used = "raw" if routes else used
        else:
            # Raw routing only
            start_lat, start_lon = float(req.start_lat), float(req.start_lon)
            end_lat, end_lon = float(req.end_lat), float(req.end_lon)
            routes = _try_route(start_lat, start_lon, end_lat, end_lon)
            used = "raw" if routes else used

        if not routes:
            raise HTTPException(404, "No route found.")

    except HTTPException:
        raise
    except nx.NetworkXNoPath:
        raise HTTPException(404, "No route found (disconnected road network between snapped points).")
    except Exception as e:
        print("[route_tool] ERROR in /route_top3")
        traceback.print_exc()
        raise HTTPException(400, str(e))

    features = []
    for r in routes:
        geom_ll = gpd.GeoSeries([r.geometry_3857], crs="EPSG:3857").to_crs("EPSG:4326").iloc[0]
        features.append(
            {
                "type": "Feature",
                "properties": {
                    "rank": r.rank,
                    "distance_m": round(r.distance_m, 2),
                    "duration_s": round(r.duration_s, 2),
                    "duration_min": round(r.duration_s / 60.0, 2),
                    "mode": mode,
                },
                "geometry": mapping(geom_ll),
            }
        )

    return {
        "type": "FeatureCollection",
        "features": features,
        "routing_endpoints": {
            "start": {"lat": float(start_lat), "lon": float(start_lon)},
            "end": {"lat": float(end_lat), "lon": float(end_lon)},
        },
        "poi_snap": snap_info,
        "used_routing": used,
    }
